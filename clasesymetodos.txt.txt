public static class operaciones{
    
    //primer metodo
    public void mensaje()
    {
        System.out.println("Soy el mejor Programador");//Método vacio
    }
    //segundo metodo
    public void nota(int numero)
    {
        if (numero >=70){
            System.out.println("APROVADO");
        }
        else
        {
            System.out.println("Reprovado");
        }
        
    }
    //tercer metodo
    public int suma(int n, int m){
        int s;
        s=n+m;
        return s;
        
    }
    //cuarto metodo
    public void numeros(int x){
        for(int i=0; i<=x;i++)
        {
            System.out.println(i);
        }
    }
    //quinto metodo
    public void summul(int q,int w, int e ){
        double t,mu;
        t=q*w;
        mu=t*e;
        System.out.println(q+"La suma de "+e+"es :"+t);
        System.out.println("por "+e+"es :"+mu);
    }
